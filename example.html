💻 代码示例

import { createBarcode, BarcodeFormat } from '@ldesign/barcode';

// 生成 EAN-13 条码
const barcode = createBarcode({
  content: '6901234567892',
  format: BarcodeFormat.EAN13,
  width: 300,
  height: 100,
  container: document.getElementById('barcode'),
});

// 下载条码
barcode.download('barcode.png');
